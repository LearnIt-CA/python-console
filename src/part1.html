import random

def initialize_game():
    """Initialize the Higher or Lower game."""
    print("Welcome to Higher or Lower!")
    print("Try to guess if the next card will be higher or lower.")
    
    # Card values from 2 to Ace (14)
    cards = list(range(2, 15))
    score = 0
    current_card = random.choice(cards)
    print(f"Starting card: {card_name(current_card)}")
    
    return cards, score, current_card

def card_name(card_value):
    """Convert numeric card value to name."""
    if card_value == 14:
        return "Ace"
    elif card_value == 13:
        return "King"
    elif card_value == 12:
        return "Queen"
    elif card_value == 11:
        return "Jack"
    else:
        return str(card_value)

def play_round(cards, current_card, guess):
    """Play a round of Higher or Lower."""
    # Draw a new card (different from current)
    new_card = random.choice([c for c in cards if c != current_card])
    print(f"New card: {card_name(new_card)}")
    
    # Check if guess is correct
    if guess == "higher" and new_card > current_card:
        return True, new_card
    elif guess == "lower" and new_card < current_card:
        return True, new_card
    else:
        return False, new_card

def main():
    cards, score, current_card = initialize_game()
    
    while True:
        # Get player's guess
        guess = input("Will the next card be higher or lower? (higher/lower/quit): ").lower()
        
        if guess == "quit":
            break
        
        if guess not in ["higher", "lower"]:
            print("Invalid input. Please enter 'higher' or 'lower'.")
            continue
        
        # Play the round
        correct, current_card = play_round(cards, current_card, guess)
        
        if correct:
            score += 1
            print(f"Correct! Your score is now {score}")
        else:
            print(f"Wrong! Game over. Final score: {score}")
            break
        
    print("Thanks for playing!")

if __name__ == "__main__":
    main()
