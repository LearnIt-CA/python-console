03_BASIC_OPERATIONS

Python supports various operations for manipulating data:

OPERATIONS 
# Arithmetic operations
a = 10
b = 3
sum_result = a + b        # Addition: 13
diff_result = a - b       # Subtraction: 7
prod_result = a * b       # Multiplication: 30
div_result = a / b        # Division: 3.3333...
floor_div = a // b        # Floor division: 3
remainder = a % b         # Modulus (remainder): 1
power = a ** b            # Exponentiation: 1000

# Comparison operations
x = 5
y = 10
print(x > y)   # Greater than: False
print(x < y)   # Less than: True
print(x >= y)  # Greater than or equal to: False
print(x <= y)  # Less than or equal to: True
print(x == y)  # Equal to: False
print(x != y)  # Not equal to: True

# Logical operations
secure = True
authorized = False
print(secure and authorized)  # Logical AND: False
print(secure or authorized)   # Logical OR: True
print(not secure)            # Logical NOT: False

# String operations
first = "Cyber"
last = "Security"
full = first + " " + last  # Concatenation: "Cyber Security"
repeated = "Secure" * 3    # Repetition: "SecureSecureSecure"
print("ber" in first)      # Membership: True

04_FUNCTIONS

Functions are reusable blocks of code that perform specific tasks:

FUNCTION DEFINITION & USAGE 
# Define a function
def greet_agent(agent_id):
    """This function greets an agent with their ID."""
    return f"Welcome, Agent {agent_id}. Your mission is ready."

# Call the function
message = greet_agent("007")
print(message)  # Output: Welcome, Agent 007. Your mission is ready.

# Function with multiple parameters
def calculate_risk(threat_level, vulnerability_score, asset_value):
    risk = (threat_level * vulnerability_score) / 10 * asset_value
    return risk

# Call function with multiple arguments
mission_risk = calculate_risk(8, 7, 1000)
print(f"Mission risk assessment: {mission_risk}")

# Function with default parameter values
def encrypt_message(message, encryption_level=3):
    # Simplified encryption simulation
    encrypted = ""
    for char in message:
        encrypted += chr(ord(char) + encryption_level)
    return encrypted

# Using the default parameter
secret1 = encrypt_message("Attack at dawn")
# Overriding the default parameter
secret2 = encrypt_message("Attack at dawn", 5)

06_CONDITIONALS

Conditional statements allow your program to make decisions:

IF-ELSE STATEMENTS 
# Basic if statement
access_level = 7
if access_level >= 5:
    print("Access granted to secure terminal")

# if-else statement
card_value = 10
player_guess = "higher"
next_card = 8

if (player_guess == "higher" and next_card > card_value) or \
   (player_guess == "lower" and next_card < card_value):
    print("Correct guess!")
else:
    print("Wrong guess!")

# if-elif-else chain (multiple conditions)
threat_level = 3

if threat_level == 1:
    print("STATUS: All clear")
elif threat_level == 2:
    print("STATUS: Caution advised")
elif threat_level == 3:
    print("STATUS: Alert - security measures activated")
else:
    print("STATUS: CRITICAL - immediate action required")

# Nested if statements
is_authenticated = True
is_authorized = True

if is_authenticated:
    print("Authentication successful")
    if is_authorized:
        print("Authorization confirmed. Access granted.")
    else:
        print("Not authorized for this resource.")

07_LOOPS

Loops execute a block of code multiple times:

LOOP STRUCTURES 
# For loop with range
print("Initiating countdown:")
for i in range(5, 0, -1):  # (start, stop, step)
    print(i)
print("Launch!")

# For loop with a list
card_faces = ["Jack", "Queen", "King", "Ace"]
for card in card_faces:
    print(f"Card: {card}")

# While loop
attempts = 0
max_attempts = 3
while attempts < max_attempts:
    print(f"Login attempt {attempts + 1} of {max_attempts}")
    # Simulated password check
    password = input("Enter password: ")
    if password == "secret":
        print("Access granted")
        break  # Exit the loop
    attempts += 1

# Loop control: break and continue
print("Scanning frequency bands...")
for freq in range(1, 11):
    if freq == 5:
        print("Restricted frequency detected, skipping...")
        continue  # Skip to the next iteration
    elif freq == 8:
        print("Enemy transmission detected. Halting scan.")
        break  # Exit the loop
    print(f"Scanning frequency {freq} GHz")

# Loop with else clause (runs if loop completes normally, without break)
for attempt in range(3):
    code = input("Enter authorization code: ")
    if code == "9843":
        print("Authorization successful")
        break
else:
    print("Maximum attempts reached. Access denied.")
